calculate_cold_war_score = {
	#分数计算-核弹
	every_country = {
		limit = {
			has_country_flag = nuke_num
		}
		set_variable = {
			nuke_score = num_of_nukes1
		}
		if = {
			limit = { 
				has_country_flag = {
					flag = nuke_num
					value = 3
				}
			}
			multiply_variable = {
				nuke_score = @nuke_triad
			}
		}
		else_if = {
			limit = { 
				has_country_flag = {
					flag = nuke_num
					value = 2
				}
			}
			multiply_variable = {
				nuke_score = @nuke_semi_triad
			}
		}
		else = {
			multiply_variable = {
				nuke_score = @nuke_regional_deployment 
			}
		}
		multiply_variable = {
			nuke_score = nuke_maintenence_ratio
		}
	}
	#分数计算-经济
	every_country = {
		limit = {
			is_in_array = {
				array = global.TFR_the_competitors
			}  
		}
		set_variable = {
			economic_score1 = gdp_billion_var
		}
		multiply_variable = {
			economic_score1 = 0.01
		}
	}
	#分数计算-军事
	every_country = {
		limit = {
			is_in_array = {
				array = global.TFR_the_competitors
			} 
		}
		set_variable = {
			military_score1 = num_armies
		}
		add_to_variable = {
			military_score1 = num_ships
		}
		add_to_variable = {
			military_score1 = num_deployed_planes
		}
	}
	#分数计算-总分
	every_country = {
		limit = {
			is_in_array = {
				array = global.TFR_the_competitors
			}  
		}
		set_variable = {
			total_score1 = military_score
		}
		add_to_variable = {
			total_score1 = economic_score
		}
		add_to_variable = {
			total_score1 = nuke_score
		}
	}

}
